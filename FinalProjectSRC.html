Part 2:Hypothesis Testing: Permutation test

# Define groups 
high_usage <- cleaned_data$TS.[cleaned_data$USG. > 25]
low_usage <- cleaned_data$TS.[cleaned_data$USG. <= 25]

# Observed difference
obs_diff <- mean(high_usage) - mean(low_usage)

# Permutation Test 
set.seed(4714)

n_permutations <- 10000
perm_diffs <- numeric(n_permutations)
all_ts <- c(high_usage, low_usage)
n_high <- length(high_usage)

for (i in 1:n_permutations) {
  shuffled <- sample(all_ts)
  perm_high <- shuffled[1:n_high]
  perm_low <- shuffled[(n_high + 1):length(all_ts)]
  perm_diffs[i] <- mean(perm_high) - mean(perm_low)
}

# Calculate P-value (two-tailed)
p_value <- mean(abs(perm_diffs) >= abs(obs_diff))
print(paste("Permutation P-value:", p_value))
## [1] "Permutation P-value: 0.0472"
# Visualize the distribution
hist(perm_diffs, main="Distribution of Permuted Differences", 
     xlab="Difference in Mean TS%", col="skyblue")
abline(v = obs_diff, col="red", lwd=2)

We will compute expected values $ E[X]=1/p$ and probabilities such as $ P(X>k) $to compare consistency between players with different usage levels.
